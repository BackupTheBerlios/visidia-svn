\subsection{Description du cas d'utilisation}

\subsubsection{Généralités}

\begin{table}[h!t]
\begin{tabular}{|p{3cm}|p{10cm}|}
\hline
Cas d'utilisation & Paramétrer le simulateur. \\ \hline
Acteurs & L'utilisateur. \\ \hline
But & Permettre à l'utilisateur de définir le comportement du simulateur. \\ \hline
Résumé Métier & L'utilisateur défini ses règles de réécriture de
manière ``graphique''. \\ \hline
Pré conditions & La partie simulation est lancée. \\ \hline
Post conditions & Le simulateur est paramétrer. \\ \hline
\multirow{2}{0cm}{Commentaires} & Le paramétrage s'applique seulement au simulateur
courant du système.\\ \hline
\end{tabular}

\end{table}
