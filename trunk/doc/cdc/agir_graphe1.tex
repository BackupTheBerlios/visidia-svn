\subsubsection{Ajouter un sommet pendant l'exécution}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}    & Ajouter un sommet pendant l'exécution\\ \hline
{Acteur}               & L'utilisateur \\ \hline
{But}                  & Ajouter un sommet dans le graphe pendant une exécution \\ \hline
{Résumé métier}        & L'utilisateur indique dans la fenêtre de simulation l'endroit où ajouter un sommet \\ \hline
{Pré-condition}        & L'utilisateur dispose d'un graphe en cours d'exécution \\ \hline
{Post-condition}       & Le sommet est ajouté au graphe \\ \hline
{Commentaires}         & Le sommet n'est pas connecté au graphe \\ \hline
\end{tabular}



\subsubsection{Supprimer un sommet pendant l'exécution}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}    & Supprimer un sommet pendant l'exécution\\ \hline
{Acteur}               & L'utilisateur \\ \hline
{But}                  & Ajouter un sommet dans le graphe pendant une exécution \\ \hline
{Résumé métier}        & L'utilisateur indique dans la fenêtre de simulation le sommet à supprimer \\ \hline
{Pré-condition}        & L'utilisateur dispose d'un graphe en cours d'exécution \\ \hline
{Post-condition}       & Le sommet et toute arête incidente à celui-ci est supprimée (voir cas d'utilisation supprimer arête) \\ \hline
{Commentaires}         & Il n'y a pas de notification du crash aux voisins. Les agents éventuellement présents sur le sommet sont supprimés \\ \hline
\end{tabular}


\subsubsection{Ajouter une arête pendant l'exécution}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}    & Ajouter une arête pendant l'exécution \\ \hline
{Acteur}               & L'utilisateur \\ \hline
{But}                  & Ajouter une arête dans le graphe pendant une exécution \\ \hline
{Résumé métier}        & L'utilisateur indique les deux sommets distincts à connecter et qui ne sont pas encore connectés \\ \hline
{Pré-condition}        & L'utilisateur dispose d'un graphe en cours d'exécution \\ \hline
{Post-condition}       & L'arête est ajoutée, \\
                       & les deux sommets à connecter prennent compte la nouvelle arête\\
		       & et l'exécution se poursuit \\ \hline
{Commentaires}         & Lors de l'ajout d'une arête, l'exécution est mise en pause \\ \hline
\end{tabular}




\subsubsection{Supprimer une arête pendant l'exécution}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}    & Supprimer une arête pendant l'exécution \\ \hline
{Acteur}               & L'utilisateur \\ \hline
{But}                  & Supprimer une arête dans le graphe pendant une exécution \\ \hline
{Résumé métier}        & L'utilisateur indique l'arête à supprimer \\ \hline
{Pré-condition}        & L'utilisateur dispose d'un graphe en cours d'exécution \\ \hline
{Post-condition}       & L'arête est supprimée, les deux sommets reliés par cette arête prennent en compte sa suppression \\ \hline
{Commentaires}         & Lors de la suppression d'une arête,
                         l'exécution est mise en pause.\\
                       & Les agents éventuellement présents sur l'arête à supprimer sont supprimés \\
 \hline 
\end{tabular}
