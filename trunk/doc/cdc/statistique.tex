
\subsubsection{Afficher des statistiques par agent}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}   & Afficher les statistiques d'un agent \\ \hline
{Acteur}              & L'utilisateur. \\ \hline
{But}                 & Afficher des statistiques pour un agent donné à la fin de l'algorithme. \\ \hline
{Résumé métier}       & L'utilisateur sélectionne un agent et voit ses statistiques : le nombre de pas qu'il a fait entre le début et la fin de l'exécution de l'algorithme, l'évolution de la taille de la mémoire de sa mémoire.\\ \hline
{Pré-condition}       & L'utilisateur sélectionne un agent.\\ \hline
{Post-condition}      & Les statistiques sur l'agent sont affichés. \\ \hline
{Commentaires}        & \\ \hline
\end{tabular}



\subsubsection{Afficher des statistiques sur l'exécution d'un algorithme}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}  & Afficher les statistiques sur l'exécution d'un algorithme \\ \hline
{Acteur}             & L'utilisateur. \\ \hline
{But}                & Afficher les statistiques sur un algorithme à la fin de son exécution \\ \hline
{Résumé métier}      & L'utilisateur sélectionne l'option
``afficher statistiques'' et consulte les informations suivantes :
coût de l'algorithme (somme des pas des agents), temps de
l'algorithme (maximum du nombre de pas des agents), nombre maximal d'agents au
cours de l'exécution.\\ \hline
{Pré-condition}      & L'algorithme a fini de s'exécuter \\ \hline
{Post-condition}     & Les statistiques sont calculées affichées \\ \hline
{Commentaires}       & \\ \hline
\end{tabular}
