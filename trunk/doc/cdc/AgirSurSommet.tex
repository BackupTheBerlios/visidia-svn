\subsubsection{Modifier la mémoire d'un sommet}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}   & Modifier la mémoire d'un sommet. \\ \hline
{Acteur}              & L'utilisateur. \\ \hline
{But}                 & Modifier la mémoire d'un sommet, c'est-à-dire son \texttt{whiteboard} et l'état de ses ports. \\ \hline
{Résumé métier}       & L'utilisateur choisit un sommet et modifie son \texttt{whiteboard} selon ses besoins. \\ \hline
{Pré-condition}       & L'utilisateur sélectionne le sommet à modifier.\\ \hline
{Post-condition}      & La mémoire du sommet sélectionné est modifiée. \\ \hline
{Commentaires}        & On ne peut modifier qu'un seul sommet à la fois.\\ \hline
\end{tabular}



\subsubsection{Éteindre un sommet}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}  & Éteindre un sommet. \\ \hline
{Acteur}             & L'utilisateur. \\ \hline
{But}                & Eteindre un sommet du réseau sans pour autant le supprimer. \\ \hline
{Résumé métier}      & L'utilisateur choisit un sommet et l'éteint. \\ \hline
{Pré-condition}      & L'utilisateur sélectionne le sommet (qui doit être allumé) à éteindre. \\ \hline
{Post-condition}     & Le sommet sélectionné est éteint. \\ \hline
{Commentaires}       & On ne peut éteindre qu'un seul sommet à la fois.\\ \hline
\end{tabular}

Scénario secondaire :
\\
\\
\begin{tabular}{|l|p{10cm}|}
\hline
{Numéro enchaînement}    & Action \\ \hline
{1}                      & L'utilisateur choisit un sommet à éteindre hébergeant un agent. \\ \hline
{2}                      & L'agent est conservé dans le sommet (et y reste). \\ \hline
{3}                      & Le système éteint le sommet. \\ \hline
\end{tabular}

\subsubsection{Consulter un sommet}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}  & Consulter un sommet. \\ \hline
{Acteur}             & L'utilisateur. \\ \hline
{But}                & Consulter la mémoire d'un sommet. \\ \hline
{Résumé métier}      & L'utilisateur choisit selon ses besoins un sommet et consulte sa mémoire. \\ \hline
{Pré-condition}      & L'utilisateur sélectionne le sommet à consulter. \\ \hline
{Post-condition}     & Le sommet sélectionné est consulté. \\ \hline

\end{tabular}




\subsubsection{Allumer un sommet}

\begin{tabular}{|l|p{10cm}|}
\hline
{Cas d'utilisation}  & Allumer un sommet.  \\ \hline
{Acteur}             & L'utilisateur. \\ \hline
{But}                & Allumer un sommet (c'est-à-dire activer à nouveau le processeur et les ports). \\ \hline
{Résumé métier}      & L'utilisateur choisit selon ses besoins un sommet et l'allume. \\ \hline
{Pré-condition}      & L'utilisateur sélectionne le sommet à allumer. Ce sommet doit être déjà éteint. \\ \hline
{Post-condition}     & Le sommet sélectionné est allumé.  \\ \hline
{Commentaires}       & On ne peut allumer qu'un seul sommet(éteind) à la fois.  \\ \hline
\end{tabular}
